Задача 1. Безопасное деление

def safe_division(a, b):
    if b == 0:
        print("На ноль делить нельзя!")
        return 0
    else:
        return a / b

print(safe_division(20, 4))   # → 5.0
print(safe_division(15, 0))   # сообщение + 0

Задача 2. Универсальная функция вычислений

def calculate(знак, x, y):
    if знак == "+": return x + y
    if знак == "-": return x - y
    if знак == "*": return x * y
    if знак == "/":
        if y == 0:
            print("На ноль нельзя!")
            return 0
        return x / y
    
    print("Я такого знака не знаю")
    return 0

print(calculate("+", 7, 8))    # 15
print(calculate("/", 10, 2))   # 5.0

Задача 3. Безопасный ввод числа

def get_number(вопрос):
    while True:
        текст = input(вопрос + " ")
        try:
            число = float(текст)
            return число
        except:
            print("Нужно написать число! Попробуй ещё раз.")

n = get_number("Сколько тебе лет?")
print("Тебе", n, "лет")

def calculator_with_history():

        действие = input("Знак (+ - * /) или 'выход': ")
        if действие == "выход":
            break
            
        x = get_number("Первое число: ")
        y = get_number("Второе число: ")
        
        if   действие == "+": res = x + y
        elif действие == "-": res = x - y
        elif действие == "*": res = x * y
        elif действие == "/": res = safe_division(x, y)
        else:
            print("Неправильный знак")
            continue
            
        строчка = f"{x} {действие} {y} = {res}"
        print(строчка)
        история.append(строчка)
    
    print("\nВся история:")
    for строка in история:
        print(строка)

Задача 3. Цена со скидкой

def calculate_discount(цена, скидка):
    if скидка < 0 or скидка > 100:
        print("Скидка 0–100%")
        return цена
        
    сколько_скидки = цена * скидка / 100
    итоговая_цена = цена - сколько_скидки
    print("Было: ", цена)
    print("Скидка:", сколько_скидки)
    print("Стало:", итоговая_цена)
    return итоговая_цена

calculate_discount(1200, 30)   # 840
