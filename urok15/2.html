# 3. –ê–Ω–∞–ª–∏–∑ —Ç–µ–∫—Å—Ç–∞
def analyze_text(text):
    vowels = "aeiouAEIOU"
    result = {
        "vowels": 0,
        "consonants": 0,
        "digits": 0,
        "other": 0
    }

    for char in text:
        if char.isdigit():
            result["digits"] += 1
        elif char.isalpha():
            if char in vowels:
                result["vowels"] += 1
            else:
                result["consonants"] += 1
        else:
            result["other"] += 1

    return result
